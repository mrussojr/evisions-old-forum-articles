Yes and yes

 

student pidm

 

SELECT SPRIDEN_PIDM, 'Dear ' || SPRIDEN_FIRST_NAME || ','
INTO :student_pidm,
     :student_intro
FROM SPRIDEN
WHERE SPRIDEN_ID = :student_id
AND SPRIDEN_CHANGE_IND IS NULL

****************************************************************

fall,spring,summer

 

SELECT DECODE(:aidy_code,
              '1617','201609',
              '1718','201709'),
       DECODE(:aidy_code,
              '1617','201702',
              '1718','201802'),
       DECODE(:aidy_code,
              '1617','201706',
              '1718','201806')
INTO :fall,
     :spring,
     :summer
FROM DUAL

 

*****************************************************************************

tot award amt - uses the variables from previous captureforms and 1 from the mapform

 

SELECT SUM(NVL(RPRATRM_OFFER_AMT,0))
INTO :tot_award_amt
FROM RPRATRM, RFRBASE
WHERE RPRATRM_PIDM = :student_pidm
AND (RPRATRM_TERM_CODE = :fall
     OR RPRATRM_TERM_CODE = :spring
     OR RPRATRM_TERM_CODE = :summer)
AND RPRATRM_FUND_CODE = RFRBASE_FUND_CODE
AND RPRATRM_AIDY_CODE = :aidy_code
GROUP BY RPRATRM_FUND_CODE, RFRBASE_PRINT_SEQ_NO
ORDER BY RFRBASE_PRINT_SEQ_NO

 

 

capture forms must be in order top to bottom  - so the 2 variables that you will use later must come first so they are fulfilled before trying to use them
