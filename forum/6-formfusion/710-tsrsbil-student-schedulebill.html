I'm doing that already but it only returns a single value(11/30 as seen in screenshot) instead of a date for each CRN.  Here is the code I am using for the Drop Date and Withdraw Date CaptureForm Queries.

 

--GetDropDate

SELECT  sfrrsts_end_date
INTO    :DropDate
FROM    sfrrsts,
                   ssbsect
WHERE   ssbsect_term_code = :Term
AND     ssbsect_crn       = :CRN
AND     sfrrsts_term_code = ssbsect_term_code
AND     sfrrsts_ptrm_code = ssbsect_ptrm_code
AND     sfrrsts_rsts_code = 'DR'

 

--GetWithdrawDate

SELECT  sfrrsts_end_date
INTO    :WithdrawDate
FROM    sfrrsts,
                   ssbsect
WHERE   ssbsect_term_code = :Term
AND     ssbsect_crn       = :CRN
AND     sfrrsts_term_code = ssbsect_term_code
AND     sfrrsts_ptrm_code = ssbsect_ptrm_code
AND     sfrrsts_rsts_code = 'DC'
