You can dynamically name the columns by writing your query using dynamic sql.  This works fine until the column names change (because an underlying change in values), at which point someone needs to run the query through validation again to get it to work.

 

Here are examples I keep on hand:

--STANDARD PIVOT, WHERE YOU WILL HARD-CODE THE COLUMN NAMES
select * from (SELECT A.EMPLOYID as 'ID', rtrim(B.LASTNAME)+', '+rtrim(B.FRSTNAME) as 'Name', B.EMPLCLAS AS 'Class', rtrim(A.APRPTOCode) as 'Code', A.APRPTORemaining as 'Remaining'
FROM prod..APR_PTO00120 A
join prod..UPR00100 B ON A.EMPLOYID = B.EMPLOYID
WHERE B.INACTIVE = '0' and A.APRPTOCode in ('PER','SICK','VAC')
GROUP BY B.EMPLCLAS, A.EMPLOYID, B.LASTNAME, B.FRSTNAME, A.APRPTOCode, A.APRPTORemaining ) as s
pivot (max(remaining) for Code in ([PER],[SICK],[VAC])) as Type order by Name

--DYNAMIC PIVOT, WHERE YOU USE A SELECT STATEMENT TO DEFINE THE COLUMN NAMES

--declare a variable for the column names
declare @cols nvarchar(max)

    --create the query to use for the column names
    select  @cols = stuff(( select distinct '],[Yr_' + FISCAL_YEAR
    from ACADEMICCALENDAR as t2
    where DATEPART(year,fiscal_year)>=datepart(year,getdate())-5
    group by FISCAL_YEAR
    having max(START_DATE)<GETDATE() and max(END_DATE)<GETDATE()
    for xml path('')), 1, 2, '') + ']'

--declare a variable to hold your query
declare @query nvarchar(max)

    --write the query, using escape characters any time you have text
    set @query = '
    select * from (
        select distinct ''Yr_''+c.FISCAL_YEAR "Fiscal_Year",A.CURRICULUM,a.PEOPLE_ID
        from ACADEMIC a
        inner join ACADEMICCALENDAR c on a.ACADEMIC_YEAR=c.ACADEMIC_YEAR and a.ACADEMIC_TERM=c.ACADEMIC_TERM and a.ACADEMIC_SESSION=c.ACADEMIC_SESSION
        where a.GRADUATED=''g''
        and a.ACADEMIC_SESSION=''main''
    ) as t1
    pivot (count(people_id) for fiscal_year in (' + @cols + ')) as p order by curriculum'
exec(@query)
