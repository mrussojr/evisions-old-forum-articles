You'll have to parse the memo field yourself.

In your database, create the type g_code_list and the function f_memo_to_table

 

CREATE OR REPLACE TYPE UH_LIB."G_CODE_LIST" AS TABLE OF VARCHAR2(100);

/

create or replace  function uh_lib.f_memo_to_table(p_memo in clob) return g_code_list
  is
    l_memo clob := p_memo || chr(13)||chr(10);
    l_comma_index PLS_INTEGER;
    l_index PLS_INTEGER := 1;
    l_tab g_code_list := g_code_list();
  BEGIN
    LOOP
      l_comma_index := INSTR(l_memo, chr(13)||chr(10), l_index);
      EXIT WHEN l_comma_index = 0; 
        if trim(SUBSTR(l_memo, l_index, l_comma_index - l_index)) is not null then
          l_tab.EXTEND;
          l_tab(l_tab.COUNT) := trim(SUBSTR(l_memo, l_index, l_comma_index - l_index));
        end if;
        l_index := l_comma_index + 2;      
    END LOOP;
    RETURN l_tab;
  END f_memo_to_table;
/  
 

On your datablock have a multiple column object for validation using the following query:

Your memo object should have multiple entry = No

 

select column_value spriden_id, 'ID does not exist' error_message from table(uh_lib.f_memo_to_table(:Memo1))
where column_value not in
(select spriden_id from spriden)
