)ACAD_STAND

ON ACAD_STAND.PIDM = MIL_DATA.PIDM

-- FOLLOWING CODE IS TO DETRMINE WHERE THE INFORMATION TO DETERMINE WHERE THE MILITARY INFORMATION COMES

COALESCE(TBVCSRV_DETAIL_CODE,SGBUSER_SUDB_CODE,SGBSTDN_DATA.RESD_CODE,STUDENT_TYPE_DATA.STYP_CODE,TEST_DATA.TEST_CODE ) ,

TBVCSRV_TERM_CODE TERM_CODE,

SGBSTDN_DATA.RESD_DESC RESD_DESC,

ON T2.TBVCSRV_PIDM = T1.SPRIDEN_PIDM

FULL OUTER JOIN SGBUSER T3

AND T3.SGBUSER_TERM_CODE = :PARM_LB_TERM

AND T3.SGBUSER_SUDB_CODE = 'Y'

FULL OUTER JOIN

(SELECT T4.SGBSTDN_PIDM PIDM,

T4.SGBSTDN_TERM_CODE_EFF TERM_CODE,

T4.SGBSTDN_RESD_CODE RESD_CODE,

STVRESD_DESC RESD_DESC

FROM SGBSTDN T4,

STVRESD

WHERE T4.SGBSTDN_TERM_CODE_EFF = :PARM_LB_TERM

AND T4.SGBSTDN_RESD_CODE IN ('A','B','D','E','F','J')

AND T4.SGBSTDN_RESD_CODE = STVRESD_CODE

) SGBSTDN_DATA

ON SGBSTDN_DATA.PIDM = T1.SPRIDEN_PIDM

FULL OUTER JOIN

(SELECT T5.SGBSTDN_PIDM PIDM,

T5.SGBSTDN_TERM_CODE_EFF TERM_CODE,

T5.SGBSTDN_STYP_CODE STYP_CODE,

STVSTYP_DESC STYP_DESC

FROM SGBSTDN T5,

STVSTYP

WHERE T5.SGBSTDN_TERM_CODE_EFF = :PARM_LB_TERM

AND T5.SGBSTDN_STYP_CODE = STVSTYP_CODE

AND T5.SGBSTDN_STYP_CODE = 'M'

) STUDENT_TYPE_DATA

ON STUDENT_TYPE_DATA.PIDM = T1.SPRIDEN_PIDM

FULL OUTER JOIN

(SELECT T6.SORTEST_PIDM PIDM,

T6.SORTEST_TESC_CODE TEST_CODE,

STVTESC_DESC TEST_DESC

FROM SORTEST T6,

STVTESC

AND T6.SORTEST_TESC_CODE = 'EXMI'

) TEST_DATA

ON TEST_DATA.PIDM = T1.SPRIDEN_PIDM

WHERE

SPRIDEN_CHANGE_IND IS NULL

AND SPRIDEN_ENTITY_IND = 'P'

AND (T2.TBVCSRV_DETAIL_CODE IS NOT NULL

OR

T3.SGBUSER_SUDB_CODE IS NOT NULL

OR

SGBSTDN_DATA.RESD_CODE IS NOT NULL

OR

STUDENT_TYPE_DATA.STYP_CODE IS NOT NULL

OR

TEST_DATA.TEST_CODE IS NOT NULL

)

) MIL_DETERMINE

ON MIL_DETERMINE.PIDM = MIL_DATA.PIDM

WHERE :PARM_BT_RUN IS NOT NULL
